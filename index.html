mixin.alert

mixin.alert(string|option) mixin.alert("消息提示，两秒后消失") var index = mixin.alert({
   content : '点击确定关闭',
   buttons :{
     '确定' : function(index,target){
       mixin.alert.close(index)
     },
     success : function(index,target){{
       console.log('弹窗弹出完成')
     },
     cancel : function(index,target){{
       console.log('弹窗完全关闭')
     }
   }
}) var index = mixin.alert({
   title : '提示',
   content : '标题提示窗',
   buttons :{
     '确定' : function(index){
       mixin.alert.close(index)
       //确定执行
     },
     '取消' : function(){
       mixin.alert.close(index)
       //取消执行
     }
   }
}) var index = mixin.alert({
   style : 'pc',
   title : '捕获页',
   content : document.getElementById("alert-blockquote").innerHTML,
   width : 'auto',
   anim : 1,
   buttons :{
     '关闭' : function(index){
       mixin.alert.close(index);
     },
   }
}) var index = mixin.alert({

mixin.tips参数option配置说明
参数类型默认值说明
contentstring|DOM|
fucntion(elem,target)		''内容，elem:触发节点，target:tips节点
widthstringauto宽度
heightstringauto高度
classNamestring''添加自定义类名或light
timenumbernull自动关闭时间（默认不关闭）
eventstring'hover'触发气泡的事件，hover：浮上，click：点击
zIndexnumber|stringnull层叠值
successfunction成功回调
closefunction关闭回调
mixin.alert参数option配置说明
参数类型默认值说明
stylestringpc移动端和PC端
titlestring''标题
contentstring|DOM''内容
widthstringauto宽度
heightstringauto高度
classNamestring''添加类名
animstring'0'动画类型0,1
shadebooleantrue遮罩层
shadeClosebooleanfalse点击遮罩层关闭弹窗
bodyScrollbooleanfalse是否关闭body的滚动条
timenumber3000自动关闭时间
closeBtnboolean''关闭按钮,false:不显示；true:显示
buttonsobject{}按钮
successfunction成功回调
cancelfunction弹窗关闭后回调
